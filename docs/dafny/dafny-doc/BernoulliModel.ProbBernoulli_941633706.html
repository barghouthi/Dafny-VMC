static function ProbBernoulli(numer: nat, denom: nat): nat ~> bool ~> (bool,nat ~> bool)
    requires denom != 0
    requires numer <= denom
    ensures forall s: RandomNumberGenerator.RNG {:trigger UniformModel.ProbUniform(denom)(s)} {:trigger f(s)} :: f(s).0 == (UniformModel.ProbUniform(denom)(s).0 < numer)
    ensures forall s: RandomNumberGenerator.RNG {:trigger UniformModel.ProbUniform(denom)(s)} {:trigger f(s)} :: f(s).1 == UniformModel.ProbUniform(denom)(s).1
    decreases numer, denom
  {
    Monad.Bind(UniformModel.ProbUniform(denom), (k: nat) => Monad.Return(k < numer))
  }
