lemma Independence.IndepFnIsCompositional
From file: Independence.dfy

{:axiom}
ghost static lemma IndepFnIsCompositional<A,B>(f: nat ~> bool ~> (A,nat ~> bool), g: A -> nat ~> bool ~> (B,nat ~> bool))
    requires IsIndepFn(f)
    requires forall a: A {:trigger g(a)} :: IsIndepFn(g(a))
    ensures IsIndepFn(Monad.Bind(f, g))
