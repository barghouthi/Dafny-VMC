ghost static predicate IsSigmaAlgebra<T(!new)>(event_space: iset<iset<T>>, sample_space: iset<T>): bool
  {
    && (forall e | e in event_space :: e <= sample_space)
    && (iset{}) in event_space
    && (forall e | e in event_space :: (sample_space - e) in event_space)
    && (forall f: nat -> iset | (forall n :: f(n) in event_space) :: (CountableUnion(f) in event_space))
  }
