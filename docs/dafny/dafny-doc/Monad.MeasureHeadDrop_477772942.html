From file: Monad.dfy

{:axiom}
ghost static lemma MeasureHeadDrop(n: nat, s: nat ~> bool)
    ensures (iset s: RandomNumberGenerator.RNG {:trigger Drop(n, s)} | Head(Drop(n, s))) in RandomNumberGenerator.event_space && RandomNumberGenerator.mu(iset s: RNG {:trigger Drop(n, s)} | Head(Drop(n, s))) == 0.5
    decreases n
