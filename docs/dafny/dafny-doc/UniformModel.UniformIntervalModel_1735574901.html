From file: Model.dfy

static function UniformIntervalModel(a: int, b: int): nat ~> bool ~> (int,nat ~> bool)
    requires a < b
    ensures forall s: RandomNumberGenerator.RNG {:trigger f(s)} :: f(s).0 == a + ProbUniform(b - a)(s).0
    decreases a, b
  {
    ProbUniformInterval(a, b)
  }
