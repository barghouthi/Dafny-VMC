function UniformPowerOfTwoModel.ProbUnifAlternative
From file: Model.dfy

static function ProbUnifAlternative(n: nat, s: nat ~> bool, k: nat := 1, u: nat := 0): (nat,nat ~> bool)
    requires k >= 1
    decreases 2 * n - k
  {
    if k > n then
      (u, s)
    else
      ProbUnifAlternative(n, Monad.Tail(s), 2*k, if Monad.Head(s) then 2*u + 1 else 2*u)
  }
