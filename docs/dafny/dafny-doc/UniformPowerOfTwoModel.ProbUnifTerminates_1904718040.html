From file: Model.dfy

{:axiom}
ghost static lemma ProbUnifTerminates(n: nat)
    requires n > 0
    ensures ghost var b: Monad.Hurd := ProbUnif(n - 1); ghost var c: nat -> bool := (x: nat) => x < n; Independence.IsIndepFn(b) && Quantifier.ExistsStar(WhileAndUntil.Helper2(b, c)) && WhileAndUntil.ProbUntilTerminates(b, c)
    decreases n
