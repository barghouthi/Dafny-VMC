function UniformPowerOfTwoModel.UnifModel
From file: Model.dfy

static function UnifModel(n: nat): nat ~> bool ~> (nat,nat ~> bool)
    ensures forall s: RandomNumberGenerator.RNG {:trigger UnifAlternativeModel(n)(s)} {:trigger f(s)} :: f(s) == UnifAlternativeModel(n)(s)
    decreases n
  {
    var f := ProbUnif(n);
    assert forall s :: f(s) == UnifAlternativeModel(n)(s) by {
      forall s ensures f(s) == UnifAlternativeModel(n)(s) {
        ProbUnifCorrespondence(n, s);
      }
    }
    f
  }
