static function ProbWhileCut<A>(c: A -> bool, b: A -> nat ~> bool ~> (A,nat ~> bool), n: nat, a: A): nat ~> bool ~> (A,nat ~> bool)
    decreases n
  {
    if n == 0 then
      Monad.Return(a)
    else (
           if c(a) then
             Monad.Bind(b(a), (a': A) => ProbWhileCut(c, b, n-1, a'))
           else
             Monad.Return(a)
         )
  }
